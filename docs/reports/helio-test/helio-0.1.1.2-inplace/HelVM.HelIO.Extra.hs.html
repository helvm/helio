   68 unfoldrM :: Monad m => (a -> m (Maybe (b, a))) -> a -> m [b]
   69 unfoldrM f a = go =<< f a where
   70   go  Nothing       = pure []
   71   go (Just (b, a')) = (b : ) <$> (go =<< f a')

   76 repeatedlyM :: Monad m => ([a] -> m (b, [a])) -> [a] -> m [b]
   77 repeatedlyM f = go where

   79   go a  = build =<< f a where build (b, a') = (b : ) <$> go a'
