   60 -- | DeConstructors
   61 safeIOToPTextIO :: Show a => IO (Safe a) -> IO Text
   62 safeIOToPTextIO a = showP <$> safeIOToIO a
   63 
   64 safeIOToIO :: IO (Safe a) -> IO a
   65 safeIOToIO a = safeToIO =<< a
   66 
   67 safeToIO :: Safe a -> IO a
   68 safeToIO = safeTToIO . liftSafe
   69 
   70 safeTToIO :: SafeT IO a -> IO a
   71 safeTToIO = liftExceptT . withExceptT (userError . errorsToString)
   72 
   73 runSafeT :: SafeT m a -> m (Safe a)
   74 runSafeT = runExceptT
   75 
   76 runSafe :: Safe a -> Safe a
   77 runSafe = id
   78 
   79 safeToText :: Safe a -> Text
   80 safeToText (Left messages) = errorsToText messages
   81 safeToText (Right       _) = ""
   82 
   83 safeToEitherLegacy :: Safe a -> EitherLegacy a
   84 safeToEitherLegacy = first errorsToString
   85 
   86 orErrorTuple :: MessageTuple -> Safe a -> a
   87 orErrorTuple t = unsafe . appendErrorTuple t
   88 
   89 orError :: Show e => e -> Safe a -> a
   90 orError e = unsafe . appendError (show e)
   91 
   92 unsafe :: Safe a -> a
   93 unsafe (Right a) = a
   94 unsafe (Left  a) = (error . errorsToText) a

   99 nonEmptyFromList message = liftSafe . maybeToSafe message . nonEmpty

  101 maybeOrError :: Show e => e -> Maybe a -> Safe a
  102 maybeOrError = maybeToSafe . show

  105 maybeToSafe = maybeToRight . D.singleton

  111 liftExceptT :: MonadError e m => ExceptT e m a -> m a
  112 liftExceptT m = liftEither =<< runExceptT m

  114 liftSafe :: MonadSafe m => Safe a -> m a
  115 liftSafe = liftEither

  131 liftMaybeOrError e = liftSafe . maybeToRight (D.singleton e)

  135 liftErrorWithTupleList m l = liftError (m <> tupleListToMessage l)

  138 liftErrorTupleList = liftError . tupleListToMessage

  140 liftErrorWithPrefix :: MonadSafe m => Message -> Message -> m a
  141 liftErrorWithPrefix prefix showed = liftErrorTuple (prefix , showed)

  143 liftErrorTuple :: MonadSafe m => MessageTuple -> m a
  144 liftErrorTuple = liftError . tupleToMessage

  146 liftError :: MonadSafe m => Message -> m a
  147 liftError = throwError . D.singleton

  149 -- | Append Message
  150 appendErrorTupleList :: MonadSafe m => [MessageTuple] -> m a -> m a
  151 appendErrorTupleList = appendError . tupleListToMessage

  154 appendErrorTuple = appendError . tupleToMessage

  156 appendError :: MonadSafe m => Message -> m a -> m a
  157 appendError message a = catchError a appendAndThrow where appendAndThrow es = throwError (es `D.snoc` message)
